aqui pueden ir bloques de codigo que puedan ser reutilizables como un boton, una barra de navegacion, cartas


Ejemplo:

// components/Button.jsx
// Con este componente tendrás un botón reutilizable en toda la app,
// solo cambia el texto pasándole la prop "label"

function Button({ label, onClick }) {
  return (
    <button onClick={onClick}>
      {label}
    </button>
  )
}

export default Button
